94@flow(name="run-batch-simulations") 

95def run_flow(context: ContextConfig, series: SeriesConfig, parameters: ParametersConfig) -> None: 

96 """Main run batch simulations flow.""" 

97 

98 if not context.working_location.startswith("s3://"): 

99 logger = get_run_logger() 

100 logger.error("Batch simulations can only be run with S3 working location.") 

101 return 

102 

103 manifest = load_dataframe(context.manifest_location, series.manifest_key) 

104 template = load_text(context.template_location, series.template_key) 

105 

106 all_job_arns: list[str] = [] 

107 

108 for group in series.groups: 

109 group_key = series.name if group is None else f"{series.name}_{group}" 

110 group_conditions = [ 

111 condition 

112 for condition in series.conditions 

113 if group is None or condition["group"] == group 

114 ] 

115 group_inits = [init for init in series.inits if group is None or init["group"] == group] 

116 

117 # Find missing conditions. 

118 missing_conditions = find_missing_conditions( 

119 manifest, series.name, group_conditions, series.seeds, series.extensions 

120 ) 

121 

122 if len(missing_conditions) == 0: 

123 continue 

124 

125 # Convert missing conditions into model input files. 

126 input_contents: list[str] = [] 

127 

128 if parameters.model.upper() == "ARCADE": 

129 condition_sets = group_template_conditions(missing_conditions, parameters.seeds_per_job) 

130 input_contents = generate_input_contents(template, condition_sets) 

131 elif parameters.model.upper() == "PHYSICELL": 

132 input_contents = render_physicell_template(template, missing_conditions, group_key) 

133 

134 if len(input_contents) == 0: 

135 continue 

136 

137 # Copy source init files to target init files. 

138 valid_seeds = {condition["seed"] for condition in missing_conditions} 

139 for init in group_inits: 

140 if len(valid_seeds.intersection(init["seeds"])) == 0: 

141 continue 

142 

143 source_key = make_key(init["name"], "inits", f"inits.{parameters.model.upper()}") 

144 source = make_key(source_key, f"{init['name']}_{init['key']}") 

145 

146 target_key = make_key(series.name, "{{timestamp}}", "inits") 

147 targets = [make_key(target_key, f"{group_key}_{seed:04d}") for seed in init["seeds"]] 

148 

149 for target in targets: 

150 for ext in init["extensions"]: 

151 copy_key(context.working_location, f"{source}.{ext}", f"{target}.{ext}") 

152 

153 # Create job definition. 

154 registry = f"{context.account}.dkr.ecr.{context.region}.amazonaws.com" 

155 job_key = make_key(context.working_location, series.name, "{{timestamp}}/") 

156 job_definition = make_batch_job( 

157 f"{context.user}_{group_key}", 

158 f"{registry}/{context.user}/{parameters.image}", 

159 context.vcpus, 

160 context.memory, 

161 [ 

162 {"name": "SIMULATION_TYPE", "value": "AWS"}, 

163 {"name": "BATCH_WORKING_URL", "value": job_key}, 

164 {"name": "FILE_SET_NAME", "value": group_key}, 

165 ], 

166 f"arn:aws:iam::{context.account}:role/BatchJobRole", 

167 ) 

168 job_definition_arn = register_batch_job(job_definition) 

169 

170 # Save input files. 

171 for index, input_content in enumerate(input_contents): 

172 input_key = make_key(series.name, "{{timestamp}}", "inputs", f"{group_key}_{index}.xml") 

173 save_text(context.working_location, input_key, input_content) 

174 

175 # Submit jobs. 

176 job_arns = submit_batch_job( 

177 group_key, 

178 job_definition_arn, 

179 context.user, 

180 context.queue, 

181 len(input_contents), 

182 ) 

183 all_job_arns = all_job_arns + job_arns 

184 

185 for job_arn in all_job_arns: 

186 exitcode = check_batch_job.with_options( 

187 retries=parameters.retries, retry_delay_seconds=parameters.retry_delay 

188 ).submit(job_arn, parameters.retries) 

189 

190 wait_for = [exitcode] 

191 

192 if parameters.terminate_jobs: 

193 terminate_status = terminate_batch_job.submit(job_arn, wait_for=wait_for) 

194 wait_for = [terminate_status] 

195 

196 if parameters.save_logs: 

197 logs = get_batch_logs.submit(job_arn, parameters.log_filter, wait_for=wait_for) 

198 log_key = make_key(series.name, "{{timestamp}}", "logs", f"{job_arn}.log") 

199 save_text.submit(context.working_location, log_key, logs) 

200 wait_for = [logs]
