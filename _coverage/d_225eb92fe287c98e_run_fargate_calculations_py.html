87@flow(name="run-fargate-calculations") 

88def run_flow(context: ContextConfig, series: SeriesConfig, parameters: ParametersConfig) -> None: 

89 """Main run fargate calculations flow.""" 

90 

91 # Check that a valid calculation type is selected. 

92 if parameters.calculate is None: 

93 logger = get_run_logger() 

94 logger.error( 

95 "No valid calculation type selected. Valid options: [ %s ]", 

96 " | ".join([member.name for member in Calculation]), 

97 ) 

98 return 

99 

100 # Check that the working location is vali.d 

101 if not context.working_location.startswith("s3://"): 

102 logger = get_run_logger() 

103 logger.error("Fargate calculations can only be run with S3 working location.") 

104 return 

105 

106 # Get the calculation type. 

107 suffix = parameters.calculate.name 

108 module_name, chunkable = parameters.calculate.value 

109 analysis_key = make_key(series.name, "analysis", f"analysis.{suffix}") 

110 

111 # Get the region suffix, if it exists. 

112 region = "" 

113 if "region" in parameters.overrides is not None: 

114 region = f"_{parameters.overrides['region']}" 

115 

116 # Create and register the task definition for the calculation. 

117 task_definition = make_fargate_task( 

118 module_name, 

119 parameters.image, 

120 context.account, 

121 context.region, 

122 context.user, 

123 context.vcpus, 

124 context.memory, 

125 ) 

126 task_definition_arn = register_fargate_task(task_definition) 

127 

128 # Import the module for the specified calculation. 

129 module = importlib.import_module(f"..{module_name}", package=__name__) 

130 

131 # Create the context and series configs for the calculation. 

132 context_config = module.ContextConfig(working_location=context.working_location) 

133 series_config = module.SeriesConfig(name=series.name) 

134 

135 for condition in series.conditions: 

136 for seed in series.seeds: 

137 series_key = f"{series.name}_{condition['key']}_{seed:04d}" 

138 

139 # If the calculation is chunkable, load results to identify chunks. 

140 if chunkable: 

141 results_key = make_key(series.name, "results", f"{series_key}.csv") 

142 results = load_dataframe(context.working_location, results_key) 

143 

144 # Check if the compiled calculation result already exists. 

145 calc_key = make_key(analysis_key, f"{series_key}{region}.{suffix}.csv") 

146 calc_key_exists = check_key(context.working_location, calc_key) 

147 

148 # If the compiled calculation result already exists, load the calculated ticks. 

149 existing_ticks = [] 

150 if calc_key_exists: 

151 existing_calc = load_dataframe(context.working_location, calc_key, usecols=["TICK"]) 

152 existing_ticks = list(existing_calc["TICK"].unique()) 

153 

154 for tick in parameters.ticks: 

155 # Skip the tick if it exists in the compiled calculation result. 

156 if tick in existing_ticks: 

157 continue 

158 

159 # Check if the individual calculation result already exists. 

160 tick_key = make_key(analysis_key, f"{series_key}_{tick:06d}{region}.{suffix}.csv") 

161 tick_key_exists = check_key(context.working_location, tick_key) 

162 

163 # Skip the tick if the individual calculation result exists. 

164 if tick_key_exists: 

165 continue 

166 

167 completed_offset_keys = [] 

168 missing_offsets_overrides = [] 

169 

170 # If the calculation is chunkable, get the completed and missing chunk offsets. 

171 if chunkable: 

172 total = results[results["TICK"] == tick].shape[0] 

173 chunk = parameters.chunk 

174 all_offsets = list(range(0, total, chunk)) if chunk is not None else [0] 

175 

176 for offset in all_offsets: 

177 if chunk is not None: 

178 offset_key = tick_key.replace( 

179 f".{suffix}.csv", 

180 f".{offset:04d}.{chunk:04d}.{suffix}.csv", 

181 ) 

182 offset_key_exists = check_key(context.working_location, offset_key) 

183 

184 if offset_key_exists: 

185 completed_offset_keys.append(offset_key) 

186 continue 

187 

188 missing_offsets_overrides.append({"offset": offset, "chunk": chunk}) 

189 else: 

190 missing_offsets_overrides.append({}) 

191 

192 # Create commands and submit (or display) tasks. 

193 for offset_overrides in missing_offsets_overrides: 

194 parameters_config = module.ParametersConfig( 

195 key=condition["key"], 

196 seed=seed, 

197 tick=tick, 

198 **offset_overrides, 

199 **parameters.overrides, 

200 ) 

201 

202 config = { 

203 "context": context_config, 

204 "series": series_config, 

205 "parameters": parameters_config, 

206 } 

207 

208 command = ["abmpipe", module_name, "::"] + make_dotlist_from_config(config) 

209 

210 if parameters.submit_tasks: 

211 submit_fargate_task.with_options(retries=2, retry_delay_seconds=1)( 

212 module_name, 

213 task_definition_arn, 

214 context.user, 

215 context.cluster, 

216 context.security_groups.split(":"), 

217 context.subnets.split(":"), 

218 command, 

219 ) 

220 else: 

221 print(" ".join(command)) 

222 

223 # If all chunk results exist, compile into unchunked result. 

224 if ( 

225 chunkable 

226 and len(completed_offset_keys) == len(all_offsets) 

227 and chunk is not None 

228 ): 

229 tick_calcs = [] 

230 

231 for key in completed_offset_keys: 

232 tick_calcs.append(load_dataframe(context.working_location, key)) 

233 

234 calc_dataframe = pd.concat(tick_calcs, ignore_index=True) 

235 save_dataframe(context.working_location, tick_key, calc_dataframe, index=False) 

236 

237 for key in completed_offset_keys: 

238 remove_key(context.working_location, key)
